#version 330

//Passed from vertex shader
in vec3 oPosition_WorldSpace;
in vec3 oPosition_ObjectSpace;
in vec3 oPosition_EyeSpace;
in vec3 oNormal_InvWorldSpace;
in vec3 oNormal_ObjectScpace;
in vec2 oUVs;
in vec3 oCamera_WorldSpace;
in float oTime;


out vec4 FragColour;

uniform sampler2D       SCT_TEXTURE2D_0;
uniform vec3            SCT_UNIFORM_FLOATIn_0;

void Sampler2D(sampler2D Texture, vec2 Texcoord, inout vec4 FinalColour)
{
 FinalColour = texture2D(Texture,Texcoord); 
}
void DirSimpleLight(vec3 normal, vec3 lightDir, inout vec4 FinalOut)
{
 
   float NdotL = max( dot(normalize(normal),normalize(lightDir)), 0.0);
   FinalOut = vec4 (1,1,1,1)*NdotL;
   
}
void MultiplyV4xV4(vec4 VecA, vec4 VecB, inout vec4 FinalColour)
{
 FinalColour = VecA*VecB; 
}


 void main()
{

vec4  OUT_FinalColour0_NF_Sampler2D_0_44;
Sampler2D(SCT_TEXTURE2D_0, oUVs, OUT_FinalColour0_NF_Sampler2D_0_44);
vec4  OUT_FinalOut0_NF_DirSimpleLight_1_12;
DirSimpleLight(oNormal_InvWorldSpace, SCT_UNIFORM_FLOATIn_0, OUT_FinalOut0_NF_DirSimpleLight_1_12);
vec4  OUT_FinalColour0_NF_MultiplyV4xV4_2_19;
MultiplyV4xV4(OUT_FinalColour0_NF_Sampler2D_0_44, OUT_FinalOut0_NF_DirSimpleLight_1_12, OUT_FinalColour0_NF_MultiplyV4xV4_2_19);
FragColour = OUT_FinalColour0_NF_MultiplyV4xV4_2_19;

}
