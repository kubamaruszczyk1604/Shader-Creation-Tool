#version 330

layout(location = 0) in vec3 vertex_position;
layout(location = 1) in vec3 vertex_colour;
layout(location = 2) in vec3 tangent;
layout(location = 3) in vec2 uvs;

uniform mat4 MVP;
uniform mat4 WORLD;

out vec3 colour;
out vec2 UVs;
out vec3 position;

void main() 
{
  colour = mat3(WORLD) * vertex_colour;//normal
  UVs = uvs;
  position = (WORLD * vec4(vertex_position,1)).xyz;
  gl_Position = MVP*vec4(vertex_position, 1.0);
  
}
